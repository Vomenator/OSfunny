section .realmodecrap
    bits 32

    enter_real_mode:
        cli
        pop esi                     ; this pops the return address from the stack into esi, which is used to return to the bootloader after setting up real mode
        lgdt [gdt16_ptr]
        jmp 0x18:.prot16     ; this jumps to the protected mode segemnt

        bits 16
    .prot16:
        mov ax, 0x20                ; this is the data segment selector, which is used to set up the data segment registers
        mov ds, ax                  ; this sets the data segment register to the data segment selector
        mov es, ax                  ; this sets the extra segment register to the data segment selector
        mov ss, ax                  ; this sets the stack segment register to the data segment selector

        mov eax, cr0                     ; this moves the value of cr0 into eax, which is used to enable protected mode
        and eax, 0xFFFFFFFE              ; this sets the PE bit in cr0
        mov cr0, eax                     ; this moves the value of eax back into cr0 enabling real mode
        jmp 0x0000:.realmode             ; this jumps to the protected mode segment

    .realmode:
        xor ax, ax                  ; this clears the ax register
        mov ds, ax                  ; this sets the data segment register to 0, which is used to set up the data segment registers
        mov es, ax                  ; this sets the extra segment register to 0, which is
        mov ss, ax                  ; this sets the stack segment register to 0, which is used to set up the stack segment register
        mov sp, 0x7000              

        ; now in real mode we set the video settings and get the fuck outta there
        mov ah, 0x00                ; this is the function number for setting the video mode, which is used to set the video mode to 640x480 with 16 bit colours
        mov al, 0x12                ; this is the video mode number for 640
        int 0x10                    ; this calls the BIOS interrupt to set the video mode

        ; back to protected mode
        lgdt [gdt32_ptr]
        mov eax, cr0
        or eax, 0x1
        mov cr0, eax

        jmp 0x08:.back32

    bits 32
    .back32:
        mov ax, 0x10
        mov ds, ax
        mov es, ax
        mov fs, ax
        mov gs, ax
        mov ss, ax
        mov esp, stack_top

        ; push return address and go back
        push esi
        ret

    gdt16:
        dq 0x0000000000000000  ; null
        dq 0x00CF9A000000FFFF  ; 32 bit code
        dq 0x00CF92000000FFFF  ; 32 bit data
        dq 0x00009A000000FFFF  ; 16 bit code
        dq 0x000092000000FFFF  ; 16 bit data

    gdt16_ptr:
        dw (gdt16_ptr - gdt16 - 1)
        dd gdt16

    gdt32_ptr:
        dw (gdt16_ptr - gdt16 - 1)
        dd gdt16
